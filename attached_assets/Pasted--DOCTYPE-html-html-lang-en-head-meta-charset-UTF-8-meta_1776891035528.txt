<script>
(async () => {
  let cryptoKey = null;
  let currentUser = '';
  let keyFingerprint = '';
 
  const $ = id => document.getElementById(id);

  async function deriveKey(passphrase) {
    const encoder = new TextEncoder();
    const rawKey = await crypto.subtle.importKey(
      'raw',
      encoder.encode(passphrase),
      'PBKDF2',
      false,
      ['deriveKey']
    );
    return crypto.subtle.deriveKey(
      { name: 'PBKDF2', salt: encoder.encode('vaultchat-v1'), iterations: 100000, hash: 'SHA-256' },
      rawKey,
      { name: 'AES-GCM', length: 256 },
      true,
      ['encrypt', 'decrypt']
    );
  }

  async function fingerprint(key) {
    const exportedKey = await crypto.subtle.exportKey('raw', key);
    const hash = await crypto.subtle.digest('SHA-256', exportedKey);
    return Array.from(new Uint8Array(hash))
      .slice(0, 10)
      .map(byte => byte.toString(16).padStart(2, '0'))
      .join(':');
  }

  async function encrypt(text) {
    const encoder = new TextEncoder();
    const iv = crypto.getRandomValues(new Uint8Array(12));
    const ciphertextBuffer = await crypto.subtle.encrypt(
      { name: 'AES-GCM', iv },
      cryptoKey,
      encoder.encode(text)
    );
    const combined = new Uint8Array(iv.length + ciphertextBuffer.byteLength);
    combined.set(iv);
    combined.set(new Uint8Array(ciphertextBuffer), 12);
    return btoa(String.fromCharCode(...combined));
  }

  async function decrypt(base64Ciphertext) {
    try {
      const combined = Uint8Array.from(atob(base64Ciphertext), x => x.charCodeAt(0));
      const iv = combined.slice(0, 12);
      const data = combined.slice(12);
      const plaintextBuffer = await crypto.subtle.decrypt(
        { name: 'AES-GCM', iv },
        cryptoKey,
        data
      );
      return new TextDecoder().decode(plaintextBuffer);
    } catch {
      return '[cannot decrypt - wrong key or corrupted data]';
    }
  }

  function escapeHTML(str) {
    return str.replace(/[&<>"']/g, char =>
      ({ '&': '&amp;', '<': '&lt;', '>': '&gt;', '"': '&quot;', "'": '&#39;' }[char])
    );
  }

  function getFormattedTime() {
    return new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
  }

  function getInitials(name) {
    return name.split(' ').map(w => w[0] || '').join('').slice(0, 2).toUpperCase() || '?';
  }

  const messages = [];

  function renderMessages() {
    const container = $('msgs');
    container.innerHTML = '';

    if (!messages.length) {
      container.innerHTML = `
        <div class="empty">
          <div class="empty-icon">
            <svg viewBox="0 0 24 24"><path d="M21 15a2 2 0 0 1-2 2H7l-4 4V5a2 2 0 0 1 2-2h14a2 2 0 0 1 2 2z"/></svg>
          </div>
          <p>No messages yet.<br>Say something securely.</p>
        </div>`;
      return;
    }

    messages.forEach(msg => {
      if (msg.isSystem) {
        const sysDiv = document.createElement('div');
        sysDiv.className = 'sys-msg';
        sysDiv.textContent = msg.text;
        container.appendChild(sysDiv);
        return;
      }

      const isMine = msg.sender === currentUser;
      const wrapDiv = document.createElement('div');
      wrapDiv.className = `msg-wrap ${isMine ? 'me' : 'them'}`;
      wrapDiv.innerHTML = `
        <div class="msg-meta">
          <div class="msg-avatar"><span>${escapeHTML(getInitials(msg.sender))}</span></div>
          <span class="msg-name">${escapeHTML(msg.sender)}</span>
          <span class="msg-time">${msg.time}</span>
        </div>
        <div class="bubble">${escapeHTML(msg.text)}</div>
        <div class="cipher-line">cipher: ${msg.cipher.slice(0, 32)}...</div>
      `;
      container.appendChild(wrapDiv);
    });

    container.scrollTop = container.scrollHeight;
  }

  $('joinBtn').addEventListener('click', async () => {
    const nameInput = $('nameInput').value.trim();
    const passInput = $('passInput').value;
    const errBox = $('errMsg');

    if (!nameInput || !passInput) {
      errBox.style.display = 'block';
      return;
    }

    errBox.style.display = 'none';
    $('joinBtn').textContent = 'Deriving key...';
    $('joinBtn').disabled = true;

    try {
      cryptoKey = await deriveKey(passInput);
      keyFingerprint = await fingerprint(cryptoKey);
      currentUser = nameInput;

      $('fpDisplay').textContent = keyFingerprint;
      $('login').style.display = 'none';
      $('chat').style.display = 'flex';

      messages.push({ isSystem: true, text: `You joined. Fingerprint: ${keyFingerprint}` });
      renderMessages();
      $('msginput').focus();
    } catch (e) {
      console.error("Key derivation failed", e);
      errBox.textContent = "Failed to generate key. Try again.";
      errBox.style.display = 'block';
    } finally {
      $('joinBtn').textContent = 'Enter room';
      $('joinBtn').disabled = false;
    }
  });

  $('passInput').addEventListener('keydown', e => { if (e.key === 'Enter') $('joinBtn').click(); });
  $('nameInput').addEventListener('keydown', e => { if (e.key === 'Enter') $('passInput').focus(); });

  async function sendMessage() {
    const inputField = $('msginput');
    const rawText = inputField.value.trim();
    if (!rawText) return;

    inputField.value = '';
    inputField.style.height = 'auto';

    const ciphertext = await encrypt(rawText);
    const decryptedText = await decrypt(ciphertext);

    messages.push({
      sender: currentUser,
      text: decryptedText,
      time: getFormattedTime(),
      cipher: ciphertext
    });

    renderMessages();
  }

  $('sendbtn').addEventListener('click', sendMessage);

  $('msginput').addEventListener('keydown', e => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      sendMessage();
    }
  });

  $('msginput').addEventListener('input', function () {
    this.style.height = 'auto';
    this.style.height = Math.min(this.scrollHeight, 120) + 'px';
  });

  $('leaveBtn').addEventListener('click', () => {
    cryptoKey = null;
    currentUser = '';
    messages.length = 0;

    $('chat').style.display = 'none';
    $('login').style.display = 'flex';
    $('passInput').value = '';
    $('nameInput').value = '';
    $('fpDisplay').textContent = '—';
  });
})();
</script>
